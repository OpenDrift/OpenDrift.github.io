    # - Weather forecast from NOAA/NCEP
    o.add_readers_from_list([
        'https://tds.hycom.org/thredds/dodsC/FMRC_ESPC-D-V02_uv3z/FMRC_ESPC-D-V02_uv3z_best.ncd',

    # Run model
